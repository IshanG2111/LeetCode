// Title: Two Sum
// URL: https://leetcode.com/problems/two-sum/
// Difficulty: Easy
// Language: python

class Solution(object):
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        hashmap={}
        for i in range(len(nums)):
            result=target-nums[i]
            if result in hashmap:
                return [hashmap[result],i]
            else:
                hashmap[nums[i]]=i
